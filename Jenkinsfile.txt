pipeline {
    agent any
    stages {
        stage("Clean Up"){
            steps {
                deleteDir()
            }
        }

        stage("terraform init") {

            steps{

                sh ("terraform init")

            }

        }
        stage("Checkout") {
            steps{
                sh "git clone https://github.com/CameronR16/DevelopmentProject.git"
            }
        }

        stage("terraform Action") {

            steps {

                echo "Terraform action is –> ${action}"
                sh ("terraform ${action} –auto-approve")
            }
        }
    }

}
