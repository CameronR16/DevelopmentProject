pipeline {
    agent any
    stages {
        stage("Clean Up"){
            steps {
                deleteDir()
            }
        }

        stage("Checkout") {
            steps{
                sh "git clone https://github.com/CameronR16/DevelopmentProject.git"
            }
        }
        stage("Switch to Cloned Repo"){
            steps{
                dir("DevelopmentProject"){
                    sh "az account list"
                    sh "terraform init"
                    sh "terraform apply --auto-approve"
                }
            }
        
        
        }
    }

}
